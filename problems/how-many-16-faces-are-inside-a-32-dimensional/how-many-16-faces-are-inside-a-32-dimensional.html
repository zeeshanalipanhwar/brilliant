The function f(n,m)f(n, m)f(n,m) can be calculated recursively. However, since there are repeated function calls with the same parameters, the intermediate results should be buffered in a data structure, so that it does not come to the stack overflow. This role is taken over by the dictionary memo. Here is the complete listing of my Python program:

memo = {}

def f(n,m):
    if n == 0 and m == 0:
        return 1
    elif m < 0 or n < 0:
        return 0
    elif (n,m) in memo:
        return memo[(n,m)]
    else:
        memo[(n,m)] = f(n-1,m-1) + 2*f(n-1,m)
        return memo[(n,m)]

print(f(32,16))

# D stores known f(n,m) values
D = np.zeros((33, 33)); D[0][0] = 1;

def f(n, m):
    if n < 0 or m < 0:
        return D[n][m]
    elif n == 0 and m == 0:
        return D[n][m]
    if not D[n][m]: # if D[n][m] is zero
        D[n][m] = f(n-1, m-1) + 2*f(n-1, m)
    return D[n][m]
